// Expanding from centers approach
class Solution {
    public String longestPalindrome(String s) {
        String ans = "";
        int n = s.length();
        for(int i=0;i<n;i++){
           // for odd length palindromic substrings
            int p1 = i, p2 = i;
            while(p1 >= 0 && p2 < n && s.charAt(p1) == s.charAt(p2)){
                if(p2-p1+1 > ans.length())  
                    ans = s.substring(p1,p2+1);
                p1--;
                p2++;
            }
             // for even length palindromic substrings
            p1 = i; p2 = i+1;
            while(p1 >= 0 && p2 < n && s.charAt(p1) == s.charAt(p2)){
                if(p2-p1+1 > ans.length())  
                    ans = s.substring(p1,p2+1);
                p1--;
                p2++;
            }
        }
        return ans;
    }
}
