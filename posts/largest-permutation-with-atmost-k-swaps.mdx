// Given an integer array A of size N consisting of unique integers from 1 to N. You can swap any two integers at most k times.
// Return the largest lexicographical value array that can be created by executing at most k swaps.

public class Solution {
    public int[] solve(int[] A, int B) {
        int n = A.length;
        int[] pos = new int[n+1];
        //storing the pos of every elmnt in A
        for(int i=0;i<n;i++) pos[A[i]] = i;

        for(int i=0;i<A.length && B > 0;i++){
            // current is in its correct pos
            if(A[i] == n-i) continue;

            // else swap it with the correct element
            // update positions in pos array
            int temp = pos[n-i];
            pos[A[i]] = pos[n-i];
            pos[n-i] = i;

            //swap
            swap(A,i,temp);
            B--;
        }
        return A;
    }
    public void swap(int[] arr,int a, int b){
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
}
