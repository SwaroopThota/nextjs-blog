public class Solution {
    public int coinchange2(int[] coins, int amount) {
    int[] dp = new int[amount + 1];
    dp[0] = 1;
    for (int coin : coins) {
      for (int x = coin; x < amount + 1; ++x) {
        dp[x] = (dp[x] + dp[x - coin]) % 1000007
      }
    }
    return dp[amount];
    }
}
