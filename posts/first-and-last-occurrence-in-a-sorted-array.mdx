public static Pair findFirstLastPosition(int[] arr, int n, int key) {
        int start = 0,end = n-1;
        int[] ans = new int[2];
        ans[0] = -1;
        ans[1] = -1;
       //finding first index
        while(start<=end){
            int mid = start + (end-start)/2;
            if(key>arr[mid]) start = mid + 1;
            else if(key<arr[mid]) end = mid-1;
            else{
                arr[0] = mid;
                end = mid - 1;
            }
        }
        start = 0;
        end = n-1;
        //finding last index
        while(start<=end){
            int mid = start + (end-start)/2;
            if(key>arr[mid]) start = mid + 1;
            else if(key<arr[mid]) end = mid-1;
            else{
                arr[1] = mid;
                end = mid - 1;
            }
        }
        return ans;
	}
