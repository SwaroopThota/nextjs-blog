// printSubstrings("abc",0,"");

void printSubstrings(String str, int index, String newString){
        if(index == str.length()){
                     System.out.println(newString+",");
                     return;
        }
        //char at index is included
        printSubstring(str, index+1, newString+str.charAt(index));
        //char at index is not included
        printSubstring(str, index+1, newString);
}
/* output:
c,
b,
bc,
a,
ac,
ab,
abc,
,
*/
