// You are given an array of integers. Sort them by frequency. In case 2 elements have the same frequency, print the smaller element first.

// 4 -2 10 12 -8 4 -> -8 -2 10 12 4 4 
// 176 -272 -272 -45 269 -327 -945 176 -> -945 -327 -45 269 -272 -272 176 176 

public class Solution {
    public class Pair{
        int key, value;
        Pair(int key,int value){
            this.key = key;
            this.value = value;
        }
        public String toString(){
            return this.key + " " + this.value + " ";
        }
    }
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int t = in.nextInt();
        while(t-- > 0){
            int n = in.nextInt();
            Map<Integer,Integer> map = new HashMap<>();
            for(int i=0;i<n;i++){
                int ele = in.nextInt();
                map.put(ele,map.getOrDefault(ele,0)+1);
            }
            Solution.Pair[] arr = new Solution.Pair[map.size()];
            int k = 0;
            for(Map.Entry<Integer,Integer> e : map.entrySet()){
                arr[k++] = new Solution().new Pair(e.getKey(),e.getValue());
            }
            Arrays.sort(arr, new Comparator<Solution.Pair>(){
                public int compare(Pair p1, Pair p2){
                    int c = Integer.compare(p1.value, p2.value);
                    if(c != 0) return c;
                    c = Integer.compare(p1.key,p2.key);
                    return c;
                }
            });
            for(Pair p : arr){
                for(int j=0;j<p.value;j++){
                    System.out.print(p.key + " ");
                }
            }
            System.out.println();
        }
    }
}
