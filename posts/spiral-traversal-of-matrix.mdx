public class Solution {
    public int[] spiralOrder(final int[][] A) {
        int n = A.length, m = A[0].length;
        int[] ans = new int[n*m];
        helper(A,0,0,n-1,m-1,ans,0);
        return ans;
    }
    public void helper(int[][] mat, int i, int j, int n, int m, int[] ans, int k){
        if(i > n || j > m) return;
        // top line
        for(int s=j   ;  k<ans.length && s<=m ;   s++)   ans[k++] = mat[i][s];
        // left line
        for(int s=i+1 ;  k<ans.length && s<=n ;   s++) ans[k++] = mat[s][m];
        // Bottom line
        for(int s=m-1 ;  k<ans.length && s>=j ;   s--) ans[k++] = mat[n][s];
        // right line
        for(int s=n-1 ;  k<ans.length && s>i  ;   s--)  ans[k++] = mat[s][j];
        helper(mat,i+1,j+1,n-1,m-1,ans,k);
    }
}
