//We use cyclic sort if we are given array to sort which contains values from range 1 to N
// Here we compare each element with its index i.e index = value -1

// index               0 1 2 3 4 5 6 7
// arr values        1 2 3 4 5 6 7 8

void cyclicSort(int[] arr){
       int i = 0;
       while(i < arr.length){
             if(arr[i] == i + 1)  // if we have the correct value at the given index, so we move to next index
                    i++;
             else            //else we swap the current value with its correct position
                    swap(arr, i, arr[i]-1);
        }
}
void swap(int[] arr, int first, int second){
        int temp = arr[first];
        arr[first] = arr[second];
        arr[second] = temp;
}
