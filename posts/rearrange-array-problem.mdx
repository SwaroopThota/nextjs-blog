// Rearrange a given array so that Arr[i] becomes Arr[Arr[i]] with O(1) extra space.
// Eg: 4, 0, 2, 1, 3 -> 3 4 2 0 1 

// Note: If we want to store two different values simultaneously -> Letâ€™s assume 2 elements are a and b, and both the elements are less than n. So if 'a' is incremented by b*n. So the element becomes a + b*n, so when a + b*n is divided by n, then the value is b and a + b*n % n is a.


public class Solution {
	void rearrange(int arr[], int n){
        // First step: Increase all values by (arr[arr[i]]%n)*n
        for (int i = 0; i < n; i++)
            arr[i] += (arr[arr[i]] % n) * n;
 
        // Second Step: Divide all values by n
        for (int i = 0; i < n; i++)
            arr[i] /= n;
    }
}


public class Solution {
	public void arrange(ArrayList<Integer> a) {
                int n = a.size();
                for(int i=0;i<n;i++)
                       a.set(i, a.get(i) + (a.get(a.get(i))%n)*n);
                 for(int i=0;i<n;i++)
                      a.set(i, a.get(i)/n);
	}
}
