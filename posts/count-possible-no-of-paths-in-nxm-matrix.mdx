//We can only move either in right(j+1) direction or down(i+1) direction one step.
//countPaths(0,0,n,m);

int countPaths(int i, int j, int n, int m){
     if(i == n-1 && j == m-1) return 1;
     int res = 0;
     if(i<n-1) res += countPaths(i+1, j, n, m);
     if(j<m-1) res += countPaths(i, j+1, n, m);
     return res;
}

//countPaths(0,0,3,3); => 6
