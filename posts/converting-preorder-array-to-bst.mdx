//[8,5,1,7,10,12]
//     ^^^   ^^(right sbtr)(>8)
//(8>)left subtre
// and call recursively
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        return helper(preorder,0,preorder.length-1);
    }
    TreeNode helper(int[] a, int s, int e){
        if(s>e) return null;
        TreeNode root= new TreeNode(a[s]);
        int i=s+1;
        while(i<=e && a[s]>a[i])
            i++;
        root.left = helper(a,s+1,i-1);
        root.right = helper(a,i,e);
        return root;
    }
}
